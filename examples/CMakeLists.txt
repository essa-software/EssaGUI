function(add_example filename)
    cmake_parse_arguments(PARSE_ARGV 1 AE "" "" "LIBS")
    string(REPLACE "/" "-" targetname "${filename}")
    essa_executable(${targetname} SOURCES ${filename}.cpp LIBS GUI ${AE_LIBS} DEV_ONLY)
endfunction()
